61  def forward(self, x, rois, roi_indices):
62  """Forward the chain.
63 
64  We assume that there are :math:`N` batches.
65 
66  Args:
67  x (Variable): 4D image variable.
68  rois (Tensor): A bounding box array containing coordinates of
69  proposal boxes. This is a concatenation of bounding box
70  arrays from multiple images in the batch.
71  Its shape is :math:`(R', 4)`. Given :math:`R_i` proposed
72  RoIs from the :math:`i` th image,
73  :math:`R' = \\sum _{i=1} ^ N R_i`.
74  roi_indices (Tensor): An array containing indices of images to
75  which bounding boxes correspond to. Its shape is :math:`(R',)`.
76 
77  """
78  # in case roi_indices is ndarray
79  roi_indices = at.totensor(roi_indices).float()
80  rois = at.totensor(rois).float()
81  indices_and_rois = t.cat([roi_indices[:, None], rois], dim=1) # 128 * (1(배경) + 4)
82  # @NOTE: important: yx->xy
83  xy_indices_and_rois = indices_and_rois[:, [0, 2, 1, 4, 3]]
84  indices_and_rois = t.autograd.Variable(xy_indices_and_rois.contiguous())
85 
86  # RoIPooling
87  pool = self.roi(x, indices_and_rois)
88  pool = pool.view(pool.size(0), -1)
89  fc7 = self.classifier(pool)
90  roi_cls_locs = self.cls_loc(fc7)
91  roi_scores = self.score(fc7)
92  return roi_cls_locs, roi_scores

181 kernel_forward = '''
182  extern "C"
183  __global__ void roi_forward(const float* const bottom_data,const float* const bottom_rois,
184  float* top_data, int* argmax_data,
185  const double spatial_scale,const int channels,const int height,
186  const int width, const int pooled_height,
187  const int pooled_width,const int NN
188  ){
189 
190  int idx = blockIdx.x * blockDim.x + threadIdx.x;
191  if(idx>=NN)
192  return;
193  const int pw = idx % pooled_width;
194  const int ph = (idx / pooled_width) % pooled_height;
195  const int c = (idx / pooled_width / pooled_height) % channels;
196  int num = idx / pooled_width / pooled_height / channels;
197  const int roi_batch_ind = bottom_rois[num * 5 + 0];
198  const int roi_start_w = round(bottom_rois[num * 5 + 1] * spatial_scale);
199  const int roi_start_h = round(bottom_rois[num * 5 + 2] * spatial_scale);
200  const int roi_end_w = round(bottom_rois[num * 5 + 3] * spatial_scale);
201  const int roi_end_h = round(bottom_rois[num * 5 + 4] * spatial_scale);
202  // Force malformed ROIs to be 1x1
203  const int roi_width = max(roi_end_w - roi_start_w + 1, 1);
204  const int roi_height = max(roi_end_h - roi_start_h + 1, 1);
205  const float bin_size_h = static_cast<float>(roi_height)
206  / static_cast<float>(pooled_height);
207  const float bin_size_w = static_cast<float>(roi_width)
208  / static_cast<float>(pooled_width);
209 
210  int hstart = static_cast<int>(floor(static_cast<float>(ph)
211  * bin_size_h));
212  int wstart = static_cast<int>(floor(static_cast<float>(pw)
213  * bin_size_w));
214  int hend = static_cast<int>(ceil(static_cast<float>(ph + 1)
215  * bin_size_h));
216  int wend = static_cast<int>(ceil(static_cast<float>(pw + 1)
217  * bin_size_w));
218 
219  // Add roi offsets and clip to input boundaries
220  hstart = min(max(hstart + roi_start_h, 0), height);
221  hend = min(max(hend + roi_start_h, 0), height);
222  wstart = min(max(wstart + roi_start_w, 0), width);
223  wend = min(max(wend + roi_start_w, 0), width);
224  bool is_empty = (hend <= hstart) || (wend <= wstart);
225 
226  // Define an empty pooling region to be zero
227  float maxval = is_empty ? 0 : -1E+37;
228  // If nothing is pooled, argmax=-1 causes nothing to be backprop'd
229  int maxidx = -1;
230  const int data_offset = (roi_batch_ind * channels + c) * height * width;
231  for (int h = hstart; h < hend; ++h) {
232  for (int w = wstart; w < wend; ++w) {
233  int bottom_index = h * width + w;
234  if (bottom_data[data_offset + bottom_index] > maxval) {
235  maxval = bottom_data[data_offset + bottom_index];
236  maxidx = bottom_index;
237  }
238  }
239  }
240  top_data[idx]=maxval;
241  argmax_data[idx]=maxidx;
242  }
243 '''
244 kernel_backward = '''
245  extern "C"
246  __global__ void roi_backward(const float* const top_diff,
247  const int* const argmax_data,const float* const bottom_rois,
248  float* bottom_diff, const int num_rois,
249  const double spatial_scale, int channels,
250  int height, int width, int pooled_height,
251  int pooled_width,const int NN)
252  {
253 
254  int idx = blockIdx.x * blockDim.x + threadIdx.x;
255  ////Importtan >= instead of >
256  if(idx>=NN)
257  return;
258  int w = idx % width;
259  int h = (idx / width) % height;
260  int c = (idx/ (width * height)) % channels;
261  int num = idx / (width * height * channels);
262 
263  float gradient = 0;
264  // Accumulate gradient over all ROIs that pooled this element
265  for (int roi_n = 0; roi_n < num_rois; ++roi_n) {
266  // Skip if ROI's batch index doesn't match num
267  if (num != static_cast<int>(bottom_rois[roi_n * 5])) {
268  continue;
269  }
270 
271  int roi_start_w = round(bottom_rois[roi_n * 5 + 1]
272  * spatial_scale);
273  int roi_start_h = round(bottom_rois[roi_n * 5 + 2]
274  * spatial_scale);
275  int roi_end_w = round(bottom_rois[roi_n * 5 + 3]
276  * spatial_scale);
277  int roi_end_h = round(bottom_rois[roi_n * 5 + 4]
278  * spatial_scale);
279 
280  // Skip if ROI doesn't include (h, w)
281  const bool in_roi = (w >= roi_start_w && w <= roi_end_w &&
282  h >= roi_start_h && h <= roi_end_h);
283  if (!in_roi) {
284  continue;
285  }
286 
287  int offset = (roi_n * channels + c) * pooled_height
288  * pooled_width;
289 
290  // Compute feasible set of pooled units that could have pooled
291  // this bottom unit
292 
293  // Force malformed ROIs to be 1x1
294  int roi_width = max(roi_end_w - roi_start_w + 1, 1);
295  int roi_height = max(roi_end_h - roi_start_h + 1, 1);
296 
297  float bin_size_h = static_cast<float>(roi_height)
298  / static_cast<float>(pooled_height);
299  float bin_size_w = static_cast<float>(roi_width)
300  / static_cast<float>(pooled_width);
301 
302  int phstart = floor(static_cast<float>(h - roi_start_h)
303  / bin_size_h);
304  int phend = ceil(static_cast<float>(h - roi_start_h + 1)
305  / bin_size_h);
306  int pwstart = floor(static_cast<float>(w - roi_start_w)
307  / bin_size_w);
308  int pwend = ceil(static_cast<float>(w - roi_start_w + 1)
309  / bin_size_w);
310 
311  phstart = min(max(phstart, 0), pooled_height);
312  phend = min(max(phend, 0), pooled_height);
313  pwstart = min(max(pwstart, 0), pooled_width);
314  pwend = min(max(pwend, 0), pooled_width);
315  for (int ph = phstart; ph < phend; ++ph) {
316  for (int pw = pwstart; pw < pwend; ++pw) {
317  int index_ = ph * pooled_width + pw + offset;
318  if (argmax_data[index_] == (h * width + w)) {
319  gradient += top_diff[index_];
320  }
321  }
322  }
323  }
324  bottom_diff[idx] = gradient;
325  }
326 '''
