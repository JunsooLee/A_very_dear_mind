8 class VOCBboxDataset:
9  """Bounding box dataset for PASCAL `VOC`_.
10 
11  .. _`VOC`: http://host.robots.ox.ac.uk/pascal/VOC/voc2012/
12 
13  The index corresponds to each image.
14 
15  When queried by an index, if :obj:`return_difficult == False`,
16  this dataset returns a corresponding
17  :obj:`img, bbox, label`, a tuple of an image, bounding boxes and labels.
18  This is the default behaviour.
19  If :obj:`return_difficult == True`, this dataset returns corresponding
20  :obj:`img, bbox, label, difficult`. :obj:`difficult` is a boolean array
21  that indicates whether bounding boxes are labeled as difficult or not.
22 
23  The bounding boxes are packed into a two dimensional tensor of shape
24  :math:`(R, 4)`, where :math:`R` is the number of bounding boxes in
25  the image. The second axis represents attributes of the bounding box.
26  They are :math:`(y_{min}, x_{min}, y_{max}, x_{max})`, where the
27  four attributes are coordinates of the top left and the bottom right
28  vertices.
29 
30  The labels are packed into a one dimensional tensor of shape :math:`(R,)`.
31  :math:`R` is the number of bounding boxes in the image.
32  The class name of the label :math:`l` is :math:`l` th element of
33  :obj:`VOC_BBOX_LABEL_NAMES`.
34 
35  The array :obj:`difficult` is a one dimensional boolean array of shape
36  :math:`(R,)`. :math:`R` is the number of bounding boxes in the image.
37  If :obj:`use_difficult` is :obj:`False`, this array is
38  a boolean array with all :obj:`False`.
39 
40  The type of the image, the bounding boxes and the labels are as follows.
41 
42  * :obj:`img.dtype == numpy.float32`
43  * :obj:`bbox.dtype == numpy.float32`
44  * :obj:`label.dtype == numpy.int32`
45  * :obj:`difficult.dtype == numpy.bool`
46 
47  Args:
48  data_dir (string): Path to the root of the training data.
49  i.e. "/data/image/voc/VOCdevkit/VOC2007/"
50  split ({'train', 'val', 'trainval', 'test'}): Select a split of the
51  dataset. :obj:`test` split is only available for
52  2007 dataset.
53  year ({'2007', '2012'}): Use a dataset prepared for a challenge
54  held in :obj:`year`.
55  use_difficult (bool): If :obj:`True`, use images that are labeled as
56  difficult in the original annotation.
57  return_difficult (bool): If :obj:`True`, this dataset returns
58  a boolean array
59  that indicates whether bounding boxes are labeled as difficult
60  or not. The default value is :obj:`False`.
61 
62  """

95  """Returns the i-th example.
96

99 
100  Args:
101  i (int): The index of the example.
102 
103  Returns:
104  tuple of an image and bounding boxes
105 
106  """
107  id_ = self.ids[i]
108  anno = ET.parse(
109  os.path.join(self.data_dir, 'dataset_laundry tag_xml', id_ + '.xml'))
110  bbox = list()
111  label = list()
112  difficult = list()
113  for obj in anno.findall('object'):
114  # when in not using difficult split, and the object is
115  # difficult, skipt it.
116  if not self.use_difficult and int(obj.find('difficult').text) == 1:
117  continue
118 
119  difficult.append(int(obj.find('difficult').text))
120  bndbox_anno = obj.find('bndbox')
121  # subtract 1 to make pixel indexes 0-based
122  bbox.append([
123  int(bndbox_anno.find(tag).text) - 1
124  for tag in ('ymin', 'xmin', 'ymax', 'xmax')])
125  name = obj.find('name').text.lower().strip()
126 
127  label.append(VOC_BBOX_LABEL_NAMES.index(name))
128 
129  bbox = np.stack(bbox).astype(np.float32)
130  label = np.stack(label).astype(np.int32)
131  # When `use_difficult==False`, all elements in `difficult` are False.
132  difficult = np.array(difficult, dtype=np.bool).astype(np.uint8) # PyTorch don't support np.bool
133 
134  # Load a image
135  img_file = os.path.join(self.data_dir, 'dataset_laundry tag', id_ + '.jpg')
136  img = read_image(img_file, color=True)
137 
138  return img, bbox, label, difficult
