123  def _suppress(self, raw_cls_bbox, raw_prob):
124  bbox = list()
125  label = list()
126  score = list()
127  # skip cls_id = 0 because it is the background class
128  for l in range(1, self.n_class):
129  cls_bbox_l = raw_cls_bbox.reshape((-1, self.n_class, 4))[:, l, :]
130  prob_l = raw_prob[:, l]
131  mask = prob_l > self.score_thresh
132  cls_bbox_l = cls_bbox_l[mask]
133  prob_l = prob_l[mask]
134  keep = non_maximum_suppression(
135  cp.array(cls_bbox_l), self.nms_thresh, prob_l)
136  keep = cp.asnumpy(keep)
137  bbox.append(cls_bbox_l[keep])
138  # The labels are in [0, self.n_class - 2].
139  label.append((l - 1) * np.ones((len(keep),)))
140  score.append(prob_l[keep])
141  bbox = np.concatenate(bbox, axis=0).astype(np.float32)
142  label = np.concatenate(label, axis=0).astype(np.int32)
143  score = np.concatenate(score, axis=0).astype(np.float32)
